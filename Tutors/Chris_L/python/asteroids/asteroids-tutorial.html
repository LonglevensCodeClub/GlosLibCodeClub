Ok. So now we want to add a few capabilities to our ship. It needs to move forward, shoot bullets and eventually collide with asteroids.  To do this we need to turn our ship into and object.

Objects are blocks of code that contain both functions and variables. This is called a class

Let’s recreate our spaceship code as a class.

class SpaceShip(Turtle):

  def __init__(self): # create spaeceship

    Turtle.__init__(self) # create spaceship using Turtle

    self.color("red")

    self.shape("ship")

 

  def doLeft(self):

    self.left(10)

 

  def doRight(self):

    self.right(10)

screenMinX = -screen.window_width()/2

class SpaceShip(Turtle):

    def __init__(self, screen, x, y):

        Turtle.__init__(self)

        self.speed(0)

        self.up()

        self.color('red')

        self.shape('myship')

        self.goto(x, y)

        self.acceleration = 0

    def fireEngine(self):

        print("Engine fired :")

        self.acceleration = self.acceleration + 1

        print(self.acceleration)

       

    def doLeft(self):

        self.left(10)

    def doRight(self):

        self.right(10)

ship = SpaceShip(screen,50,-50)

screen.onkey(ship.doLeft,'Left')

screen.onkey(ship.doRight,'Right')

screen.onkey(ship.fireEngine,'Up')

screen.listen()
